document.addEventListener('DOMContentLoaded', function () {
            const csrfToken = "{{ csrf_token }}";
            const sendMailButton = document.getElementById('send_mail_button');

            // Add a click event listener to the "Send Mail" button
            sendMailButton.addEventListener('click', function (event) {
                event.preventDefault(); // Prevent the default form submission

                // Extract the user's email input value
                const email = document.getElementById('id_email').value;

                // Send an AJAX request to your server to initiate the email sending process
                const request_body = {
                    method: "POST",
                    headers: {
                        'Content-Type': 'application/json',
                        'X-CSRFToken': csrfToken
                    },
                    body: JSON.stringify({
                        email: email
                    })
                };

                fetch('http://127.0.0.1:8000/send_email/', request_body)
                    .then(response => response.json())
                    .then(data => {
                        // Handle the response data as needed (e.g., show success message or errors)
                        console.log(data);
                        if (data.success) {
                            // Redirect to a success page or show a success message
                            alert("Email sent successfully. Check your inbox for further instructions.");
                        } else {
                            // Display an error message
                            const errorElement = document.querySelector('.error-message');
                            errorElement.innerHTML = 'Failed to send email. Please try again later.';
                        }
                    })
                    .catch(error => {
                        // Handle errors from the fetch request
                        console.error('Error:', error);
                    });
            });
        });
